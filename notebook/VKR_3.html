In [8]:
model_smn = tf.keras.Sequential(
    [
        normalizer,
        layers.Dense(16, activation="tanh"),
        layers.Dense(8, activation="relu"),
        layers.Dense(8, activation="relu"),
        layers.Dense(1),
    ]
)

model_smn.compile(
    optimizer=tf.keras.optimizers.Adam(0.001),
    loss="mean_squared_error",
    metrics=[tf.keras.metrics.RootMeanSquaredError()],
)
